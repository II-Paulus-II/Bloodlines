## First set of effects are country scope 

pass_down_achaemenid_effect = {
    every_character = {
        limit = {
            OR = {
                    exists = father
                    exists = mother
            }
            any_parent = {
                has_trait = achaemenid
            }
            NOT = {
                has_trait = achaemenid
            }
        }
        force_add_trait = achaemenid
    }  
}
